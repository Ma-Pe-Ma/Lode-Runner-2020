#version 450 core
layout (location = 0) in vec2 aPos;
layout (location = 1) in vec2 aTexturePos;

out vec2 TexCoord;
uniform vec2 gPos;

vec2 abs;
vec2 pos;

void main() {
		
	pos.x = (6 * (aPos.x + gPos.x) - 93) / 140;
	pos.y = (2 * (aPos.y - gPos.y) + 27) / 28;
	
	TexCoord = aTexturePos;
	gl_Position = vec4(pos.x, pos.y, 0.0, 1.0);
}
